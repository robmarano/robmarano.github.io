\section*{Chapter 4 Retrospective: Architecture Evolution}

\begin{frame}{Chapter 4 Retrospective: Architecture Evolution}
\begin{itemize}
    \item \textbf{Single-Cycle Datapath:} One instruction per clock cycle. Cycle time is dictated by the longest path (Load Word). Simple control, but poor clock frequency.
    \item \textbf{Multi-Cycle Datapath:} Broke instructions into 5 steps (Fetch, Decode, Execute, Memory, Write-Back). Faster clock, but complex FSM control.
    \item \textbf{Pipelined Processor:} Overlapped the 5 execution stages to achieve an ideal CPI of 1.0. Massive throughput gains but introduced structural, data, and control hazards.
    \item \textbf{Hazard Resolution:} Implemented Forwarding Units, Hazard Detection Units (stalls/bubbles), and Branch Prediction (flushing).
    \item \textbf{Exceptions \& Interrupts:} Added EPC and Cause registers, expanding the datapath to handle unpredictable external/internal events.
\end{itemize}
\end{frame}

\section{1. Introduction to Memory Hierarchy}

\begin{frame}{1. The Journey: From Processor to Memory}
\begin{itemize}
    \item \textbf{Where we've been:} We built the logic gates, the ALU, the single-cycle datapath, and finally the pipelined processor with hazard detection.
    \vspace{0.3cm}
    \item \textbf{The Final Frontier:} A fast CPU is useless if it has no data. We must now design the Memory System.
    \vspace{0.3cm}
    \item \textbf{The Historical Problem:} Early computers had simple, flat memory. Over time, processors became exponentially faster, but memory speeds lagged. CPUs were starving for data!
    \vspace{0.3cm}
    \item \textbf{The Solution:} Architects invented the \textbf{Memory Hierarchy} (Caches, Main Memory, Virtual Memory) to bridge this chasm.
\end{itemize}
\end{frame}

\begin{frame}{1. The Processor/Memory Paradox}
\begin{itemize}
    \item Computer architects face a fundamental physical paradox:
    \item \textbf{Processors are incredibly fast} (Clock cycles in nanoseconds or picoseconds).
    \item \textbf{Memory is relatively slow} (Access times in tens of nanoseconds or milliseconds).
    \item If the CPU has to wait for Main Memory on every instruction fetch and data load, the massive speed gains of pipelining ($CPI \approx 1.0$) are entirely lost!
\end{itemize}
\end{frame}

\begin{frame}{1. The Principle of Locality}
\begin{itemize}
    \item To keep the processor fed with data without stalling, we rely on the \textbf{Principle of Locality}:
    \vspace{0.5cm}
    \item \textbf{Temporal Locality (Locality in Time):}
    \begin{itemize}
        \item If an item is referenced, it will tend to be referenced again soon.
        \item \textit{Example:} Loops iterating over the same variables.
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Spatial Locality (Locality in Space):}
    \begin{itemize}
        \item If an item is referenced, items whose addresses are close by will tend to be referenced soon.
        \item \textit{Example:} Sequentially accessing elements in an array.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{1. The Memory Hierarchy Pyramid}
\begin{center}
    \includegraphics[height=5.5cm,keepaspectratio]{../../../../../Image Bank/ch005-9780128201091/jpg-9780128201091/005001.jpg}
\end{center}
\begin{itemize}
    \item We exploit locality by implementing multiple tiers of memory.
    \item \textbf{Closer to CPU}: Faster, smaller, more expensive per bit.
    \item \textbf{Further from CPU}: Slower, larger, cheaper per bit.
\end{itemize}
\end{frame}

\begin{frame}{1. Key Terminology}
\begin{itemize}
    \item \textbf{Block (or Line):} The minimum unit of information that can be either present or not present in a two-level hierarchy.
    \item \textbf{Hit:} Data requested by the processor appears in some block in the upper level.
    \begin{itemize}
        \item \textbf{Hit Rate:} The fraction of memory accesses found in the upper level.
        \item \textbf{Hit Time:} The time required to access data in the upper level.
    \end{itemize}
    \item \textbf{Miss:} Data is not found in the upper level.
    \begin{itemize}
        \item \textbf{Miss Penalty:} The time required to fetch the block from the lower level into the upper level, plus the time to deliver it to the processor.
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{1. Harris Perspective: The ``Desk'' Analogy}
\begin{itemize}
    \item \textit{Harris \& Harris} provide a highly physical analogy for the memory hierarchy:
    \vspace{0.3cm}
    \item \textbf{The CPU}: You, sitting at a desk, actively working on a problem.
    \item \textbf{The Cache}: The physical surface of your desk. Small, but you can access any paper on it instantly (SRAM).
    \item \textbf{Main Memory}: A filing cabinet across the room. It holds much more data, but retrieving a folder requires you to stand up and walk over (DRAM Miss Penalty).
    \item \textbf{Magnetic Disk}: The warehouse down the street. It holds everything, but takes a massive amount of time to fetch.
\end{itemize}
\end{frame}

\begin{frame}{1. Average Memory Access Time (AMAT)}
\begin{itemize}
    \item How do we mathematically measure the performance of this hierarchy?
    \vspace{0.3cm}
    \item \textbf{AMAT Formula:} \texttt{Hit Time + (Miss Rate $\times$ Miss Penalty)}
    \vspace{0.3cm}
    \item \textit{Example}: If cache hit time is 1 cycle, miss penalty is 100 cycles, and miss rate is 5\%:
    \item \texttt{AMAT = 1 + (0.05 $\times$ 100) = 1 + 5 = 6 cycles}
\end{itemize}
\end{frame}

\section{2. Memory Technologies}

\begin{frame}{2. Memory Technologies: SRAM vs DRAM}
\begin{itemize}
    \item \textbf{SRAM (Static RAM):}
    \begin{itemize}
        \item Used for caches (L1, L2, L3).
        \item Fast (0.5ns - 2.5ns access time).
        \item Requires no refreshing to keep data (hence ``static'').
        \item Low density and very expensive (6 transistors per bit).
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{DRAM (Dynamic RAM):}
    \begin{itemize}
        \item Used for Main Memory.
        \item Slower (50ns - 70ns access time).
        \item Data stored as a charge on a capacitor; must be periodically ``refreshed'' because charge leaks.
        \item High density and cheaper (1 transistor per bit).
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{2. Deep-Dive: Inside a DRAM Cell}
\begin{itemize}
    \item \textbf{1T1C Architecture:} 1 Transistor + 1 Capacitor per bit. Massive density.
    \vspace{0.3cm}
    \item \textbf{Destructive Reads:} Reading the capacitor drains its tiny electrical charge. The sense amplifiers must immediately rewrite the data back into the cell.
    \vspace{0.3cm}
    \item \textbf{Refresh Cycles:} Capacitors leak! The memory controller must halt the CPU and \textit{refresh} (read and rewrite) every row every $\approx 64$ms to prevent data corruption.
    \vspace{0.3cm}
    \item \textbf{Multiplexing:} Row Address Strobe (RAS) and Column Address Strobe (CAS) send the row and column halves of the address over the exact same pins to save space on the physical chip.
\end{itemize}
\end{frame}

\begin{frame}{2. Non-Volatile Memory}
\begin{itemize}
    \item \textbf{Flash Memory:}
    \begin{itemize}
        \item Non-volatile semiconductor memory (EEPROM).
        \item Survives power-off.
        \item Reads are fast, but writes are significantly slower and physically wear out the memory cells over time (requires wear leveling).
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Magnetic Disk:}
    \begin{itemize}
        \item Used for massive secondary storage.
        \item Relies on moving mechanical parts (platters and read/write heads).
        \item Measured in milliseconds (millions of times slower than SRAM).
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{2. Hamacher Perspective: Write Policies}
\begin{itemize}
    \item When the CPU executes a \texttt{sw} (Store Word), data is written to the Cache. But when does it update Main Memory?
    \vspace{0.3cm}
    \item \textbf{Write-Through:} Data is written to \textit{both} the cache block and the lower-level memory simultaneously. Perfectly safe, but extremely slow.
    \vspace{0.3cm}
    \item \textbf{Write-Back:} Data is written \textit{only} to the cache. The block is marked with a \textbf{Dirty Bit}. It is only written to Main Memory if that block is evicted later. Incredibly fast, but vulnerable to power loss.
\end{itemize}
\end{frame}

\begin{frame}{2. Hamacher Perspective: Memory Interleaving}
\begin{itemize}
    \item To reduce the \textbf{Miss Penalty} when fetching a block from slow DRAM, \textit{Hamacher} introduces \textbf{Interleaving}.
    \vspace{0.3cm}
    \item Instead of storing sequential blocks on one memory chip, the memory controller distributes consecutive blocks across multiple discrete memory banks.
    \vspace{0.3cm}
    \item The CPU can send fetch requests to all banks simultaneously, receiving the data in parallel rather than sequentially!
\end{itemize}
\end{frame}

\section{3. The Basics of Caches}

\begin{frame}{3. Mapping Memory to Cache}
\begin{itemize}
    \item The \textbf{Cache} is the level of the memory hierarchy closest to the CPU.
    \item How do we map memory blocks from the massive Main Memory into the tiny Cache?
    \item There are three fundamental topologies, trading off hardware complexity against miss rates:
    \begin{enumerate}
        \item \textbf{Direct-Mapped Caches}
        \item \textbf{Fully Associative Caches}
        \item \textbf{Set-Associative Caches (N-Way)}
    \end{enumerate}
\end{itemize}
\end{frame}

\begin{frame}{3. Direct Mapped Caches}
\begin{itemize}
    \item In a direct-mapped cache, each memory block maps to exactly \textit{one} specific block in the cache.
    \item \textbf{Mapping Formula:} \texttt{(Block address) modulo (Number of blocks in the cache)}
    \vspace{0.3cm}
    \item \textbf{Address Bit Breakdown:}
    \begin{itemize}
        \item \textbf{Offset:} The lowest bits. Identifies specific byte. Width = $\log_2(\text{Bytes per Block})$.
        \item \textbf{Index:} The middle bits. Identifies exact cache row. Width = $\log_2(\text{Number of Cache Blocks})$.
        \item \textbf{Tag:} Leftover upper bits stored alongside data to verify a match.
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Pros/Cons:} Extremely fast hit time (only one tag to check). Very cheap. Prone to high \textit{Conflict Misses}.
\end{itemize}
\end{frame}

\begin{frame}{3. Fully Associative Caches}
\begin{itemize}
    \item To solve conflict misses, a \textbf{Fully Associative} cache allows any memory block to be placed in \textit{any} available cache block.
    \vspace{0.3cm}
    \item \textbf{Mapping:} There is no Index field! The physical address only contains the \textbf{Tag} and \textbf{Offset}.
    \item \textbf{Hardware Cost:} The cache must compare the requested Tag against \textit{every single Tag in the cache simultaneously}. Requires massive, expensive parallel hardware comparators.
    \vspace{0.3cm}
    \item \textbf{Pros/Cons:} Lowest possible miss rate. Incredibly expensive and consumes significant power. Restricted to very small memory structures (like TLBs).
\end{itemize}
\end{frame}

\begin{frame}{3. Set-Associative Caches (N-Way)}
\begin{itemize}
    \item The ``Goldilocks'' compromise. The cache is divided into ``Sets'', and each set contains $N$ cache blocks (e.g., 2-way, 4-way, 8-way).
    \item \textbf{Mapping:} A memory block maps to exactly \textit{one} Set (using the Index), but can be placed into \textit{any} of the $N$ blocks within that set.
    \vspace{0.3cm}
    \item \textbf{Address Bit Breakdown:}
    \begin{itemize}
        \item \textbf{Offset:} $\log_2(\text{Bytes per Block})$.
        \item \textbf{Index:} $\log_2(\text{Number of Sets})$.
        \item \textbf{Tag:} Leftover bits.
    \end{itemize}
    \vspace{0.3cm}
    \item \textbf{Pros/Cons:} Greatly reduces conflict misses compared to Direct-Mapped, without the insane hardware costs of Fully Associative. Uses LRU replacement.
\end{itemize}
\end{frame}

\section{4. SystemVerilog Hardware Implementations}

\begin{frame}[fragile]{4. Level 1: Basic Synchronous Memory}
\begin{itemize}
    \item How do we build Main Memory in RTL? As a massive array of registers.
\end{itemize}
\begin{verbatim}
module main_memory (
    input  logic        clk,
    input  logic        we,       // Write Enable
    input  logic [31:0] addr,
    input  logic [31:0] wd,       // Write Data
    output logic [31:0] rd        // Read Data
);
    logic [31:0] mem [0:1023];    // 4KB Memory Array

    assign rd = mem[addr[31:2]];  // Asynchronous Read (Word aligned)

    always_ff @(posedge clk) begin
        if (we) mem[addr[31:2]] <= wd; // Synchronous Write
    end
endmodule
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{4. Level 2: Direct-Mapped Cache Arrays}
\begin{itemize}
    \item A Direct-Mapped Cache requires three distinct physical arrays:
\end{itemize}
\begin{verbatim}
// Parameters for a small 4KB Direct-Mapped Cache
// Block Size = 1 Word (4 Bytes)
// Number of Blocks = 1024 

logic [31:0] cache_data  [0:1023]; // The actual data blocks
logic [19:0] cache_tags  [0:1023]; // The upper 20 bits of the address
logic        cache_valid [0:1023]; // 1-bit valid flags
\end{verbatim}
\end{frame}

\begin{frame}[fragile]{4. Level 3: Combinational Hit/Miss Logic}
\begin{itemize}
    \item The Cache Controller must slice the address and determine a Hit.
\end{itemize}
\begin{verbatim}
    // Step 1: Slice the 32-bit CPU physical address
    assign index = cpu_address[11:2];  // 10 bits for 1024 blocks
    assign tag   = cpu_address[31:12]; // Remaining 20 upper bits

    // Step 2: Extract the stored Tag and Valid bit using the Index
    assign stored_tag = cache_tags[index];
    assign is_valid   = cache_valid[index];

    // Step 3: Hit Logic (Combinational)
    assign cache_hit = is_valid & (stored_tag == tag);
    
    // If cache_hit == 0, trigger the FSM to stall and fetch from DRAM!
\end{verbatim}
\end{frame}

\section{5. Worked Textbook Examples}

\begin{frame}{5. Example 1: Address Fields (Harris \& Harris)}
\textbf{Problem:} 32-bit address, Direct-Mapped Cache, 32 KB capacity, 64-Byte blocks. Calculate bits for Offset, Index, and Tag.
\begin{itemize}
    \item \textbf{1. Offset:} Driven by Block Size.
    \[ \text{Offset Bits} = \log_2(64) = 6 \text{ bits} \]
    \item \textbf{2. Index:} Driven by Number of Blocks.
    \[ \text{Blocks} = \frac{32,768 \text{ Bytes}}{64 \text{ Bytes}} = 512 \text{ Blocks} \]
    \[ \text{Index Bits} = \log_2(512) = 9 \text{ bits} \]
    \item \textbf{3. Tag:} Driven by the remaining bits.
    \[ \text{Tag Bits} = 32 - 9 - 6 = 17 \text{ bits} \]
\end{itemize}
\end{frame}

\begin{frame}{5. Example 2: Access Trace (Patterson \& Hennessy)}
\textbf{Problem:} Direct-Mapped Cache with 8 blocks. Access sequence: 22, 26, 22, 26, 16, 3, 16, 18. 

\begin{table}[]
\begin{tabular}{|l|l|l|l|}
\hline
\textbf{Access} & \textbf{Block Address} & \textbf{Cache Index (Block \% 8)} & \textbf{Hit/Miss} \\ \hline
1               & 22                     & $22 \pmod 8 = 6$                    & Miss              \\ \hline
2               & 26                     & $26 \pmod 8 = 2$                    & Miss              \\ \hline
3               & 22                     & $22 \pmod 8 = 6$                    & Hit               \\ \hline
4               & 26                     & $26 \pmod 8 = 2$                    & Hit               \\ \hline
5               & 16                     & $16 \pmod 8 = 0$                    & Miss              \\ \hline
6               & 3                      & $3 \pmod 8 = 3$                     & Miss              \\ \hline
7               & 16                     & $16 \pmod 8 = 0$                    & Hit               \\ \hline
8               & 18                     & $18 \pmod 8 = 2$                    & Miss (Evicts 26)  \\ \hline
\end{tabular}
\end{table}
\textbf{Result:} 5 Misses, 3 Hits.
\end{frame}

\begin{frame}{5. Example 3: Performance Impact (Hamacher)}
\textbf{Problem:} Base CPI = 1.0, Hit Time = 1 cycle, Miss Penalty = 100 cycles, Miss Rate = 5\%.
\begin{itemize}
    \item \textbf{1. Calculate Average Memory Access Time (AMAT):}
    \[ \text{AMAT} = \text{Hit Time} + (\text{Miss Rate} \times \text{Miss Penalty}) \]
    \[ \text{AMAT} = 1 + (0.05 \times 100) = 1 + 5 = 6 \text{ cycles} \]
    \item \textbf{2. Calculate Effective CPI:}
    \[ \text{Memory Stall Cycles} = \text{Miss Rate} \times \text{Miss Penalty} = 5 \]
    \[ \text{Effective CPI} = \text{Base CPI} + \text{Memory Stall Cycles} = 1.0 + 5.0 = 6.0 \]
\end{itemize}
\textbf{Conclusion:} A 5\% miss rate makes the CPU 6 times slower!
\end{frame}

\begin{frame}{Synthesis: Quantifying Performance (Ch 1-4)}
\textbf{1. The Baseline: Iron Law (Ch 1)}
\[ \text{CPU Time} = \text{IC} \times \text{CPI} \times \text{Clock Cycle Time} \]
\begin{itemize}
    \item \textbf{Single-Cycle:} Ideal CPI $= 1.0$, but Clock Cycle Time is massive.
\end{itemize}
\vspace{0.3cm}
\textbf{2. The Pipelining Upgrade (Ch 4)}
\[ \text{CPI}_{\text{pipeline}} = \text{Ideal CPI (1.0)} + \text{Pipeline Stall Cycles / Instr} \]
\begin{itemize}
    \item Clock cycle is drastically reduced (split into 5 stages).
    \item CPI increases slightly due to structural, data, and control hazards.
    \item \textbf{Proof:} The massive clock speed increase vastly outweighs the minor stall penalty.
\end{itemize}
\end{frame}

\begin{frame}{Synthesis: The Memory Wall (Ch 5)}
\textbf{3. The Memory Wall \& AMAT}
Main memory is slow. If we access it every cycle, the pipeline breaks.
\[ \text{AMAT} = \text{Hit Time} + (\text{Miss Rate} \times \text{Miss Penalty}) \]
\vspace{0.3cm}
\textbf{4. Fully Integrated System Performance}
We calculate the exact time penalty of cache misses on our pipeline:
\[ \text{Mem Stalls} = \text{IC} \times \frac{\text{Mem Accesses}}{\text{Instr}} \times \text{Miss Rate} \times \text{Miss Penalty} \]
\[ \text{CPI}_{\text{actual}} = \text{CPI}_{\text{pipeline}} + \frac{\text{Mem Stall Cycles}}{\text{Instr}} \]
\textbf{Conclusion:} Caches successfully exploit locality to keep Miss Rates $<5\%$, driving AMAT close to 1 cycle and preserving our pipelined clock rate!
\end{frame}
